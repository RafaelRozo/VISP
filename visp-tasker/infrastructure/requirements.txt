# =============================================================================
# VISP/Tasker — Infrastructure Tooling Requirements
# =============================================================================
# This file documents the CLI tools and versions required to manage
# the VISP/Tasker infrastructure. Install these before running any
# Terraform or Docker commands.
#
# NOTE: These are system-level tools, not Python packages.
#       Install via your system package manager or official installers.
# =============================================================================

# ---------------------------------------------------------------------------
# Terraform — Infrastructure as Code
# Install: https://developer.hashicorp.com/terraform/install
# ---------------------------------------------------------------------------
# terraform >= 1.5.0
#
# macOS:   brew install terraform
# Linux:   See https://developer.hashicorp.com/terraform/install
# Verify:  terraform version

# ---------------------------------------------------------------------------
# AWS CLI — AWS API interactions
# Install: https://docs.aws.amazon.com/cli/latest/userguide/getting-started-install.html
# ---------------------------------------------------------------------------
# aws-cli >= 2.0
#
# macOS:   brew install awscli
# Linux:   curl "https://awscli.amazonaws.com/awscli-exe-linux-x86_64.zip" -o "awscliv2.zip"
# Verify:  aws --version
# Config:  aws configure (set region to us-east-1)

# ---------------------------------------------------------------------------
# Docker — Container runtime
# Install: https://docs.docker.com/get-docker/
# ---------------------------------------------------------------------------
# docker >= 24.0
#
# macOS:   Install Docker Desktop from https://www.docker.com/products/docker-desktop
# Linux:   See https://docs.docker.com/engine/install/
# Verify:  docker --version

# ---------------------------------------------------------------------------
# Docker Compose — Multi-container orchestration
# (Included with Docker Desktop; install separately on Linux)
# ---------------------------------------------------------------------------
# docker compose >= 2.20 (V2 plugin, not standalone docker-compose)
#
# Verify:  docker compose version

# ---------------------------------------------------------------------------
# Optional but Recommended
# ---------------------------------------------------------------------------

# tflint — Terraform linter
# Install: https://github.com/terraform-linters/tflint
# macOS:   brew install tflint
# Verify:  tflint --version

# tfsec — Terraform security scanner
# Install: https://github.com/aquasecurity/tfsec
# macOS:   brew install tfsec
# Verify:  tfsec --version

# jq — JSON processor (useful for inspecting Terraform outputs)
# macOS:   brew install jq
# Verify:  jq --version

# ---------------------------------------------------------------------------
# Quick Setup (macOS with Homebrew)
# ---------------------------------------------------------------------------
# brew install terraform awscli docker tflint tfsec jq

# ---------------------------------------------------------------------------
# Usage
# ---------------------------------------------------------------------------
# Initialize Terraform:
#   cd infrastructure/terraform
#   terraform init
#
# Plan deployment (dev):
#   terraform plan -var-file=environments/dev.tfvars
#
# Apply deployment (dev):
#   terraform apply -var-file=environments/dev.tfvars
#
# Local development:
#   cd infrastructure/docker
#   cp .env.example .env  # Edit with your values
#   docker compose up -d
#
# Run tests:
#   cd infrastructure/docker
#   docker compose -f docker-compose.yml -f docker-compose.test.yml up --abort-on-container-exit
